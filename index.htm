using System;
using System.Text;

namespace myApp
{
  class Program
  {
    static void Main(string[] args)
    {
      bool error = false;

      if (error)
      {
        Console.WriteLine("Error: Error");
        Environment.Exit(1);
      }
      else if (!error)
      {
        Console.WriteLine("Press any key to exit.");
        Console.ReadKey();
        Environment.Exit(0);
      }
      else
      {
        Console.WriteLine("Error: An unknown error has occurred.");
        Environment.Exit(1);
      }
    }
  }
}
